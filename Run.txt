git init
                                        "  فى المشروع  ليتحكم  فى ترجمة ورفع الملفات  git يعمل على اضافة ملف  الـ"
<!-- Git remote-->
git remote add origin (link)
                                        " برابط الريبو الحقيقى link  واستبدل  git اضافة رابط الريبو المراد  التحكم بها من خلال  "

                                        " لعرض روابط الريبو المتحكم بها من خلال  "

                                        "  اضافة جميع الملفات الموجوده فى فولدر الجهاز إلى منطقة الانتظار فى وضع الرفع  "
git add (add file name)
                                        "  fil name اضافة ملف محدد إلى منطقة الانتظار ويجب كتابة اسم الملف او الفولدر بدلا من كلمة "
<!-- Git status-->
git status
                                        " Git معرفة حالة التواقف فى  "
<!-- Git commit-->
git commit -m "(add commit)"
                                        " إضافة التعليق على ما تم تنفيذه ويفضل كتابة ما تم  تعديله على المشرووع "
<!-- Git push-->
git push -u origin (add name baranch)
                                        " رفع الملفات على المستودع المراد مع كتابة اسم المستودع المراد الرفع عليه ام الرئيسى او الفرعي"
<!-- Git branch-->
git branch
                                        "لعرض جميع الفروع الفرعية والاساسية الموجوده فى الريبو "
git branch  (add Name of new branch)
                                        "  إنشاء فرع جديد فى الريبو للعمل عليه على حدي دون الذهاب اليه"
git checkout -b (add Name of new branch)
                                        " إنشاء فرع جديد فى الريبو والذهاب اليه للعمل عليه  على حدي"
git checkout (add name branch)
                                        " التنقل من الفرع الحالي الى فرع اخر من الموجودين فى الريبو "
git branch -d (add name branch)
                                        " حذف فرع من الفرع الموجودين فى الريبو "
git merge (add name branch)
                                        "دمج  الفرع الحالي مع فرع اخر من الموجودين فى الريبو "
<!-- Git log-->
git log
                                        "لعرض جميع سجلات التعليقات التى تمت على الفرع الحالي  "
git log --oneline
                                        " لعرض جميع سجلات التعليقات التى تمت على الفرع الحالي بشكل مختصر ومنظم "
git log --oneline --decorate --all --graph
                                        " لعرض جميع سجلات التعليقات التى تمت على جميع الفروع  بشكل مختصر ومنظم "
<!-- Git rm-->
git rm (add name file)
                                        "لحذف ملف من ملفات المشروع   "
<!-- Git stash-->
git stash (add name file)
                                        "حفظ التعديلات مؤقتاً بدون رفع لسهولة التنقل الى  فرع أخر دون فقد  التعديلات  " 
<!-- Git diff-->                                               
git diff
                                        "لعرض التغييرات بين منطقة العمل الحالية ,التعليق الأخير   "
<!-- Git stash-->
git stash
                                        " حفظ مؤقتاً للتغيرات لسهولة التنقل بين الفروع دون فقد التغيرات  "
